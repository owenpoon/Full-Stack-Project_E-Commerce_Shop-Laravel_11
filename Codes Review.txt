5. To edit AuthAdmin.php
--> use Illuminate\Support\Facades\Auth; /* To import Auth */

--> public function handle(Request $request, Closure $next): Response{
        if(Auth::check()){
            if(Auth::user()->utype==='ADM'){
                return $next($request);
            }
            else {
                Session::flush();
                return redirect()->route('login');
            }
        }
        else {
            return redirect()->route('login');
        }
    }

15. To modify on 0001_01_01_000000_create_users_table.php

--> Schema::create('users', function (Blueprint $table) {
            $table->id();
            $table->string('name');
            $table->string('email')->unique();

            $table->string('mobile')->unique(); /*Add mobile*/

            $table->timestamp('email_verified_at')->nullable();
            $table->string('password');
            $table->string('utype')->default('USR')->comment('ADM for admin & USR for User or Customer');
            $table->rememberToken();
            $table->timestamps();
    });

11. To modify on RegisterController.php

--> protected function validator(array $data){
        return Validator::make($data, [
            'name' => ['required', 'string', 'max:255'],
            'email' => ['required', 'string', 'email', 'max:255', 'unique:users'],
            'mobile' => ['required', 'digits:10', 'unique:users'], /*Add mobile*/
            'password' => ['required', 'string', 'min:8', 'confirmed'],
        ]);
    }

--> protected function create(array $data){
        return User::create([
            'name' => $data['name'],
            'email' => $data['email'],
            'mobile' => $data['mobile'],  /*Add mobile*/
            'password' => Hash::make($data['password']),
        ]);
    }
